# Execute One

class ProgrammingLanguage :
    def __init__ (self, name):
        self.name = name


python = ProgrammingLanguage("Python")
nodejs = ProgrammingLanguage("Node Js")

print(python)
print(python.name)
print(nodejs)
print(nodejs.name)
